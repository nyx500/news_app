def convertToTfidfVectors(X_train, X_val):
    """
        Convert text data into TF-IDF featuresets.
        
        Input Parameters:
        X_train (list of str): a list of text strings for training model on
        X_val (list of str): a list of text strings for validation set for testing

        Output:
        tfidf_vectorizer (scikit-learn TfidfVectorizer instance)
        tfidf_train: vectorized representation of the inputted list of training texts
        tfidf_test: vectorized representation of the inputted list of validation texts
    """
    # Instantiate TF-IDF vectorizer with max doc frequency as in the DataFlair article and English stop words
    tfidf_vectorizer = TfidfVectorizer(stop_words='english', max_df=0.7)
    tfidf_train = tfidf_vectorizer.fit_transform(X_train)
    tfidf_test = tfidf_vectorizer.transform(X_val)
    return tfidf_train, tfidf_test

# Define a reusable function for evaluating classifier performance
def classifierEvaluation(y_val, y_pred, dataset_name, average_type="macro", cmap="viridis"):
  """
    Input Parameters:
      y_val (list): list of ground truth fake news labels (0 = real, 1 = fake)
      y_pred (list): list of predicted labels
      average_type (str): determines whether to use macro or micro evaluation metrics, macro is better when classes are imbalanced to gauge performance on minority
      cmap (str): matplotlib colormap for Seaborn-generated confusion matrix
    Output:
      dictionary storing scores for 4 evaluation metrics
  """

  # Calculate the basic metrics
  accuracy = accuracy_score(y_val, y_pred)
  precision = precision_score(y_val, y_pred, average=average_type) # use macro due to class imbalance, to represent performance across all classes esp. minority class
  recall = recall_score(y_val, y_pred, average=average_type)
  macro_f1 = f1_score(y_val, y_pred, average=average_type)

  # Print the results
  print(f'Accuracy: {accuracy:.4f}')
  print(f'Precision (Macro): {precision:.4f}')
  print(f'Recall (Macro): {recall:.4f}')
  print(f'Macro F1-score: {macro_f1:.4f}')

  # Print the classification report
  print("\nClassification Report:\n")
  print(classification_report(y_val, y_pred))

  # Generate a confusion matrix
  c_matrix = confusion_matrix(y_val, y_pred)

  # Visualize confusion matrix using Seaborn heatmap
  plt.figure(figsize=(6, 5)) # Set figure size
  sns.heatmap(c_matrix, annot=True, fmt='d', cmap=cmap, # fmt = 'd' means output integers for the counts
            xticklabels=["real", "fake"], yticklabels=["real", "fake"]) # Map 0 = real, 1 = fake
  plt.title(f"Confusion Matrix for Fake News Detection: {dataset_name}")
  plt.xlabel("Predicted Class")
  plt.ylabel("Ground Truth")
  plt.show()

  return {'accuracy': accuracy, 'precision': precision, 'recall': recall, 'macro_f1': macro_f1}

def trainPAModel(tfidf_train, y_train):
    """
        Train the Passive Aggressive Classifier.
        Input Parameters:
            tfidf_train: training text data vectorized using basic TF-IDF vectorizer
            y_train (list of int): labels (0 = real, 1 = fake) for train data
        Output:
            pac (PassiveAggressiveClassifier): the trained PA classifier
    """
    pac = PassiveAggressiveClassifier(max_iter=50)
    pac.fit(tfidf_train, y_train)
    return pac

def evaluatePACModel(model, tfidf_val, y_val, dataset_name):
    """
        Evaluate the model and calculate accuracy and confusion matrix.
        Input Parameters:
            model: a trained Passive Aggressive Classifier model
            tfidf_val: vectorized validation set text samples
            y_val: labels for validation set text data
    """
    y_pred = model.predict(tfidf_val)
    eval_dict = classifierEvaluation(y_val, y_pred, dataset_name) # use macro-F1 averages due to class imbalance
    return eval_dict
